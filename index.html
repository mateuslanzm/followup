# Classe com herança
class Funcionario:
    def __init__(self, nome, cargo, salario):
        self.nome    = nome
        self.cargo   = cargo
        self.salario = salario

    def reajuste(self, percentual):
        self.salario *= (1 + percentual / 100)

    def __repr__(self):
        return f"Funcionario({self.nome!r}, R${self.salario:.2f})"


class Desenvolvedor(Funcionario):
    def __init__(self, nome, salario, linguagens):
        super().__init__(nome, "Desenvolvedor", salario)
        self.linguagens = linguagens


dev = Desenvolvedor("Mateus", 8000, ["Python", "SQL"])
dev.reajuste(10)
print(dev)  # Funcionario('Mateus', R$8800.00)
